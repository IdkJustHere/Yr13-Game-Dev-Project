Week 1: Camera Rig & Isometric Projection Mathematics

Scene: scenes/components/isometric_camera_rig.tscn

This scene will be instantiated in every level. The rig handles the mathematical conversion between screen-space mouse coordinates and world-space grid coordinates.

IsometricCameraRig (Node3D) ├── CameraArm (Node3D) [Position: (0, 0, 0)] │ └── Camera3D [Position: (20, 20, 20), Looking at origin] │ └── Projection: Orthogonal │ └── Size: 10.0 (adjust for zoom level) │ └── Near: 0.1, Far: 1000.0 │ └── Cull Mask: Layers 1, 2, 3 (not UI) ├── RayCastOrigin (Marker3D) [Position: matches Camera3D] └── CameraController (Script)

Script: scripts/systems/camera_controller.gd

Attach to IsometricCameraRig node. Implements discrete 90-degree rotations around the Y-axis at 45-degree increments (isometric angles).

Isometric Angle Reference:
0 degrees (NE): Camera at (20, 20, -20) looking at origin
90 degrees (SE): Camera at (-20, 20, -20)
180 degrees (SW): Camera at (-20, 20, 20)
270 degrees (NW): Camera at (20, 20, 20)
Increment stored as int 0-3, multiplied by 90 for rotation.

Appendix: Comprehensive Code Reference

A. Grid Coordinate Mathematics

# grid_manager.gd - Complete implementation reference extends Node const GRID_SIZE := 1.0 # meters per cell var cells: Dictionary = {} # Vector3i -> CellData class CellData: var conveyor: Node3D = null var item: Node3D = null var is_blocked: bool = false func world_to_grid(world_pos: Vector3) -> Vector3i: # Floor ensures negative coordinates round correctly return Vector3i( floor(world_pos.x / GRID_SIZE), 0, # Assuming flat plane at y=0 floor(world_pos.z / GRID_SIZE) ) func grid_to_world(grid_pos: Vector3i) -> Vector3: # Center of cell return Vector3( grid_pos.x * GRID_SIZE + GRID_SIZE/2, 0.2, # Slightly above floor grid_pos.z * GRID_SIZE + GRID_SIZE/2 ) func get_cell(grid_pos: Vector3i) -> CellData: if not cells.has(grid_pos): cells[grid_pos] = CellData.new() return cells[grid_pos] func is_valid_placement(grid_pos: Vector3i) -> bool: if is_out_of_bounds(grid_pos): return false var cell = get_cell(grid_pos) return cell.conveyor == null and not cell.is_blocked

B. Conveyor Direction Logic

# conveyor_logic.gd extends Node3D class_name ConveyorLogic enum Orientation { NORTH = 0, EAST = 1, SOUTH = 2, WEST = 3 } enum Type { STRAIGHT, CORNER_LEFT, CORNER_RIGHT } @export var orientation: Orientation = Orientation.NORTH @export var type: Type = Type.STRAIGHT const VECTORS = [ Vector3i(0, 0, -1), # North Vector3i(1, 0, 0), # East Vector3i(0, 0, 1), # South Vector3i(-1, 0, 0) # West ] func get_output_direction() -> Vector3i: match type: Type.STRAIGHT: return VECTORS[orientation] Type.CORNER_LEFT: # Turn left: N->W, E->N, S->E, W->S var new_orient = (orientation + 3) % 4 return VECTORS[new_orient] Type.CORNER_RIGHT: # Turn right: N->E, E->S, S->W, W->N var new_orient = (orientation + 1) % 4 return VECTORS[new_orient] return Vector3i.ZERO func rotate_clockwise(): orientation = (orientation + 1) % 4 rotation_degrees.y = orientation * 90

C. Item Movement Interpolation

# item.gd extends Node3D class_name ConveyorItem @export var move_speed: float = 2.0 # cells per second (inversely related to tick) var current_grid: Vector3i var target_grid: Vector3i var is_moving: bool = false var move_t: float = 0.0 func start_move(to_grid: Vector3i): target_grid = to_grid is_moving = true move_t = 0.0 # Visual rotation to face movement var dir = target_grid - current_grid if dir != Vector3i.ZERO: look_at(GridManager.grid_to_world(target_grid)) func _process(delta): if is_moving: move_t += delta * move_speed if move_t >= 1.0: # Arrived position = GridManager.grid_to_world(target_grid) current_grid = target_grid is_moving = false move_t = 0.0 emit_signal("arrived", target_grid) else: # Interpolate var start = GridManager.grid_to_world(current_grid) var end = GridManager.grid_to_world(target_grid) position = start.lerp(end, ease(move_t, 0.2)) # Slight ease

D. Jam Detection Algorithm (Simultaneous Resolution)

# movement_controller.gd extends Node signal jam_detected(position: Vector3i, items: Array) signal movements_resolved(movements: Dictionary) func resolve_movements(): var proposed: Dictionary = {} # target_pos -> item var jams: Array = [] # Phase 1: Collect intentions for item in ItemManager.get_items(): if not item.is_moving and item.current_grid: var conveyor = GridManager.get_conveyor_at(item.current_grid) if conveyor: var next = item.current_grid + conveyor.get_output_direction() # Check bounds if not GridManager.is_within_bounds(next): continue # Check collision with static objects if GridManager.is_blocked(next): continue # Check collision with other items already there if GridManager.get_item_at(next): jams.append({"pos": next, "items": [item, GridManager.get_item_at(next)]}) continue # Check if another item wants to move here if proposed.has(next): jams.append({"pos": next, "items": [item, proposed[next]]}) else: proposed[next] = item # Phase 2: Handle jams if jams.size() > 0: for jam in jams: jam_detected.emit(jam.pos, jam.items) GameManager.register_jam() # Phase 3: Execute valid moves (if game not over) if GameManager.jam_count < GameManager.max_jams: for target_pos in proposed.keys(): var item = proposed[target_pos] item.start_move(target_pos) GridManager.move_item(item.current_grid, target_pos) movements_resolved.emit(proposed)

# scoring_system.gd extends Node class_name ScoringSystem static func calculate_medal(level_data: LevelData, stats: LevelStats) -> int: # Returns: 0=FAIL, 1=BRONZE, 2=SILVER, 3=GOLD if not stats.quota_met: return 0 # FAIL var time_threshold = level_data.par_time * 0.5 # 50% time remaining for speed var efficiency_threshold = level_data.max_tiles * 0.5 # 50% tiles for Gold var is_efficient = stats.tiles_used <= efficiency_threshold var is_fast = stats.time_remaining >= time_threshold if is_efficient and is_fast: return 3 # GOLD elif is_fast: return 2 # SILVER else: return 1 # BRONZE

Appendix A: Grid Coordinate System

# Isometric Grid Conversion Reference
func world_to_grid(world_pos: Vector3) -> Vector2i:
    var x = world_pos.x / CELL_SIZE
    var z = world_pos.z / CELL_SIZE
    return Vector2i(round(x), round(z))

func grid_to_world(grid_pos: Vector2i) -> Vector3:
    return Vector3(
        grid_pos.x * CELL_SIZE, 
        0, 
        grid_pos.y * CELL_SIZE
    )

Appendix B: Conveyor Logic

class_name ConveyorTile extends Node3D

enum Type { STRAIGHT, CORNER_LEFT, CORNER_RIGHT, SPLITTER }
enum Direction { NORTH, EAST, SOUTH, WEST }

@export var type: Type = Type.STRAIGHT
@export var facing: Direction = Direction.NORTH

func get_input_directions() -> Array[Direction]:
    match type:
        Type.STRAIGHT: return [(facing + 2) % 4]  # Opposite
        Type.CORNER_LEFT: return [(facing + 1) % 4]  # Perpendicular
        _: return []

func get_output_direction() -> Direction:
    return facing

Appendix C: Item Movement

# In TickManager.gd - Handles discrete movement
func process_tick():
    var moves = []  # Array of {item, from, to}
    
    # Phase 1: Calculate all intended moves
    for cell in GridManager.get_occupied_cells():
        var tile = GridManager.get_tile(cell)
        if tile.has_item() and tile.can_output():
            var next_cell = cell + dir_to_vector(tile.get_output_direction())
            if GridManager.can_accept_item(next_cell):
                moves.append({item: tile.item, from: cell, to: next_cell})
    
    # Phase 2: Execute moves (prevent swap deadlock)
    for move in moves:
        execute_move(move)

Appendix D: Direction Routing

# Determines next cell based on tile type and current direction
func get_exit_direction(tile_type: String, entry_dir: Vector2i) -> Vector2i:
    match tile_type:
        "straight":
            return entry_dir
        "corner_left":
            return entry_dir.rotated(-PI/2) # 90 deg left
        "corner_right":
            return entry_dir.rotated(PI/2)  # 90 deg right
    return Vector2i.ZERO
